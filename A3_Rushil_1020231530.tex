    \[f_{k+1} \circ (f_k \circ \cdots \circ f_1)\]

    \[(f_k \circ \cdots \circ f_1)^{-1} \circ f_{k+1}^{-1}\]

    \[f_1^{-1} \circ \cdots \circ f_k^{-1} \circ f_{k+1}^{-1}\]

    \[f_1^{-1} \circ \cdots \circ (f_k^{-1} \circ f_{k+1}^{-1})\]
